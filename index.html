// 最终版：Leaflet 真3D建筑（完美适配你带height的GeoJSON） const map = L.map('map').setView([28.68, 112.88], 16); L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map); let buildings = {}; let floodData = {}; // 加载3D建筑（核心：按height拉伸） fetch('buildings.geojson') .then(res => res.json()) .then(geoData => { // 用Leaflet的3D插件，直接按height拉伸 const layer = L.geoJSON(geoData, { style: (feature) => ({ color: '#000', weight: 1, fillColor: 'white', fillOpacity: 0.8, // 3D拉伸：按height字段设置高度 renderer: L.canvas(), extrude: feature.properties.height || 10 }) }).addTo(map); // 绑定建筑ID和高度 layer.eachLayer(layer => { const id = layer.feature.properties.id; const height = layer.feature.properties.height; buildings[id] = layer; // 点击弹窗 layer.bindPopup(`

建筑ID: ${id}

状态：安全

水深：0 米

建筑高度：${height} 米

`); }); // 加载洪水数据 fetch('flood_animation_data.json') .then(res => res.json()) .then(data => { floodData = data; const times = Object.keys(data); const slider = document.getElementById('timeSlider'); const label = document.getElementById('timeLabel'); slider.max = times.length - 1; label.textContent = times[0]; slider.oninput = () => { const t = times[slider.value]; label.textContent = t; updateBuildings(floodData[t]); }; updateBuildings(floodData[times[0]]); }); }); // 建筑变色逻辑 function updateBuildings(list) { list.forEach(b => { const layer = buildings[b.id]; if (!layer) return; let color = 'white'; let status = '安全'; if (b.s === 1) { color = 'yellow'; status = '受威胁'; } if (b.s === 2) { color = 'red'; status = '淹没'; } layer.setStyle({ fillColor: color }); layer.setPopupContent(`

建筑ID: ${b.id}

状态：${status}

水深：${b.d} 米

`); }); } // 【关键】开启3D视角，直接看到立体效果 map.getRenderer()._canvas.style.transform = 'perspective(1200px) rotateX(65deg)'; map.getRenderer()._canvas.style.transformOrigin = 'center center';
